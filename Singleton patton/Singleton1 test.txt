//싱글톤 이해를 위해 test한 코딩

class Singleton 				//싱글톤 클래스
{
private:
	Singleton() {}; 			//프라이빗 생성자
	static Singleton* instance;		//스테틱 클래스 포인터 인스턴스 변수;
public:
	static Singleton* GetInstance(int num)	//스테틱 클래스 포인터 함수 인스턴스 호출시 리턴한다!
	{
		cout << num;
		return instance;		//인스턴트 리턴을 통해 객체 생성
	}
};

Singleton* Singleton::instance = nullptr; 	//static 클래스포인터 인스턴트를 nullptr로 초기화 합니다.

int main(void)
{
	Singleton::GetInstance(7);		//클래스를 통한 접근을 시도해야 합니다.
	return 0;
}
